        // Suas funções originais mantidas intactas
        function textToInvisible(text) {
            return [...text].map(c => c.charCodeAt(0).toString(2).padStart(8, '0'))
                .join('').replace(/0/g, '\u200b').replace(/1/g, '\u200c');
        }

        function invisibleToText(text) {
            let bin = text.replace(/\u200b/g, '0').replace(/\u200c/g, '1');
            let res = "";
            for(let i=0; i<bin.length; i+=8) res += String.fromCharCode(parseInt(bin.slice(i, i+8), 2));
            return res;
        }

        function assinar() {
            let t = document.getElementById('texto').value;
            let payload = btoa(JSON.stringify({id: crypto.randomUUID(), hash: "SHA256_placeholder"}));
            document.getElementById('texto').value = textToInvisible(payload) + t;
            alert("Assinado! Copie o texto.");
        }

        function verificar() {
            let t = document.getElementById('texto').value;
            let inv = t.replace(/[^\u200b\u200c]/g, '');
            try {
                let decoded = atob(invisibleToText(inv));
                alert("Dados extraídos: " + decoded);
            } catch(e) {
                alert("Nenhuma assinatura válida encontrada.");
            }
        }
    </script>
